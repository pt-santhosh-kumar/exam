Program 6: 
Data Pre-Processing 
import pandas as pd
from sklearn.preprocessing import MinMaxScaler
df = pd.DataFrame({
    'id':[1,2,3,4],
    'f1':[10,20,None,40],
    'f2':[100,200,300,None],
    'cat':['A','B','A','B']
})
print("Original Data:\n", df)
df = pd.melt(df, id_vars=['id','cat']).pivot_table( index=['id','cat'], 
columns='variable', values='value' ).reset_index()
df[['f1','f2']] = df[['f1','f2']].fillna(df[['f1','f2']].mean())
df = df[df['f1'] > 15]
df = pd.merge(df, pd.DataFrame({'id':[1,2,3,4],'extra':[5,6,7,8]}), on='id')
df[['f1','f2']] = MinMaxScaler().fit_transform(df[['f1','f2']])
print("\nFinal Data:\n", df)

Program 8: 
Linear Regression 
import numpy as np, matplotlib.pyplot as plt
from sklearn.datasets import fetch_california_housing
from sklearn.linear_model import LinearRegression
X, y = fetch_california_housing(return_X_y=True)
p = LinearRegression().fit(X, y).predict(X)
print("MAE:", round(abs(y-p).mean(),2), "RMSE:", round(np.sqrt(((y-p)**2).mean()),2))
plt.scatter(y, p)
plt.plot([min(y), max(y)], [min(y), max(y)], 'r--')
plt.show()
Program 9: 
Logistic Regression 
import matplotlib.pyplot as plt
from sklearn.datasets import load_breast_cancer
from sklearn.linear_model import LogisticRegression
from sklearn.metrics import *
X, y = load_breast_cancer(return_X_y=True)
p = LogisticRegression(max_iter=5000).fit(X,y).predict(X)
print("Accuracy:", round(accuracy_score(y,p),2))
plt.scatter(X[:,0], X[:,1], c=y)
plt.show()
cm = confusion_matrix(y,p)
plt.imshow(cm, cmap='Blues')
for i in range(2):
    for j in range(2): plt.text(j,i,cm[i,j])
plt.colorbar()
plt.show()
Program 10: 
Decision Tree Classifier 
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.tree import DecisionTreeClassifier, plot_tree
from sklearn.metrics import accuracy_score
X,y = load_iris(return_X_y=True)
Xtr,Xte,ytr,yte = train_test_split(X,y)
m = DecisionTreeClassifier().fit(Xtr,ytr)
print("Accuracy:", accuracy_score(yte,m.predict(Xte)))
plot_tree(m)
plt.show()
Program 11: 
K-Nearest Neighbours (KNN) Algorithm 
import matplotlib.pyplot as plt
from sklearn.datasets import load_iris
from sklearn.model_selection import train_test_split
from sklearn.neighbors import KNeighborsClassifier
from sklearn.metrics import *
X,y = load_iris(return_X_y=True)
Xtr,Xte,ytr,yte = train_test_split(X,y,test_size=0.3)
p = KNeighborsClassifier().fit(Xtr,ytr).predict(Xte)
print("Accuracy:", round(accuracy_score(yte,p),2))
print(classification_report(yte,p))
cm = confusion_matrix(yte,p)
plt.imshow(cm,cmap='Blues')
for i in range(3):
    for j in range(3): plt.text(j,i,cm[i,j])
plt.colorbar()
plt.show()
Program 12: 
K-Means Clustering 
import matplotlib.pyplot as plt
from sklearn.datasets import load_wine
from sklearn.cluster import KMeans
from sklearn.preprocessing import StandardScaler
X,_ = load_wine(return_X_y=True)
X = StandardScaler().fit_transform(X)
k = KMeans(3).fit(X)
plt.scatter(X[:,0],X[:,1],c=k.labels_)
plt.scatter(*k.cluster_centers_[:,:2].T,marker='X',c='red',label = "Centroids")
plt.legend()
plt.show()
